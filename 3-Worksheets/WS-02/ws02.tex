\question
Given that
	\begin{align*}
	\lim_{x \ra 2} f(x) = 5 , \quad \lim_{x \ra 2} g(x) = 2 \quad \text{ and } \quad \lim_{x \ra 2} h(x) = -2 ,
	\end{align*}
find the following limits. If you can't use one of the limit rules, explain why.
\begin{multicols}{2}
	\begin{parts}
	\part[2]
	$\lim_{x \ra 2} (f(x) + 5g(x))$.
	
	\part[2]
	$\lim_{x \ra 2} (g (x))^3$.
	
	\part[2]
	$\lim_{x \ra 2} \frac{3f(x)}{g(x)}$.
	
	\part[2]
	$\lim_{x \ra 2} ( f(x) g(x))$.
	
	\part[2]
	$\lim_{x \ra 2} \frac{g(x)}{2 + h(x)}$.
	\end{parts}
\end{multicols}
